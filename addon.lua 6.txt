-- SUPER SIMPLE ANTI-AFK - Chỉ 10 dòng code
local VirtualInput = game:GetService("VirtualInputManager")

spawn(function()
    while true do
        task.wait(55) -- Chờ 55 giây (an toàn < 60s)
        
        -- Gửi input ẩn
        VirtualInput:SendKeyEvent(true, Enum.KeyCode.F24, false, nil)
        task.wait(0.01)
        VirtualInput:SendKeyEvent(false, Enum.KeyCode.F24, false, nil)
        
        print("[Anti-AFK] Input sent at " .. os.date("%H:%M:%S"))
    end
end)

print("Anti-AFK đang chạy - Gửi input mỗi 55 giây")
